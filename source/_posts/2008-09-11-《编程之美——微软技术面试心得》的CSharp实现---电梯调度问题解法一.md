```
class Program
{
    static void Main(string[] args)
    {
        int[] person = {0, 1, 1, 1, 1, 1, 15}; //总共六层楼，没有0层
        int needed = 0;
        int min = 0;
        int target = 0;
        for (int i = 1; i <= 6; i++)
        {
            needed = 0;
            for (int j = 1; j < i; j++)
            {
                //目的在i层以下的
                needed += person[j] * (i - j);
            }

            for (int j = i + 1; j <= 6; j++)
            {
                //目的在i层上面的
                needed += person[j] * (j - i);
            }

            if (min == 0 || min >= needed)
            {
                min = needed;
                target = i;
            }
        }

        Console.WriteLine(target);
        Console.Read();
    }
```
